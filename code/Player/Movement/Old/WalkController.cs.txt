		var maxs = new Vector3( +girth, +girth, height ) * GameObject.Parent.Transform.Scale;

		SetBBox( mins, maxs );
	}

	public virtual float GetWishSpeed()
	{
		var ws = -1;// Duck.GetWishSpeed();
		if ( ws >= 0 ) return ws;

		if ( Input.Down( "Duck" ) || IsDucking ) return CrouchSpeed;

	public virtual void WalkMove()
	{

		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		var wishdir = WishVelocity.Normal;
		var wishspeed = WishVelocity.Length;

		WishVelocity = WishVelocity.WithZ( 0 );
		WishVelocity = WishVelocity.Normal * wishspeed;

		pl.Velocity = pl.Velocity.WithZ( 0 );
		Accelerate( wishdir, wishspeed, 0, Acceleration );
		pl.Velocity = pl.Velocity.WithZ( 0 );

		//   Player.SetAnimParam( "forward", Input.Forward );
		//   Player.SetAnimParam( "sideward", Input.Right );
		//   Player.SetAnimParam( "wishspeed", wishspeed );
		//    Player.SetAnimParam( "walkspeed_scale", 2.0f / 190.0f );
		//   Player.SetAnimParam( "runspeed_scale", 2.0f / 320.0f );

		//  DebugOverlay.Text( 0, Pos + Vector3.Up * 100, $"forward: {Input.Forward}\nsideward: {Input.Right}" );

		// Add in any base velocity to the current velocity.
		pl.Velocity += pl.BaseVelocity;

		try
		{
			if ( pl.Velocity.Length < 1.0f )
			{
				pl.Velocity = Vector3.Zero;
				return;
			}

			// first try just moving to the destination
			var dest = (pl.Position + pl.Velocity * Time.Delta).WithZ( pl.Position.z );

			var pm = TraceBBox( pl.Position, dest );

			if ( pm.Fraction == 1 )
			{
				pl.Position = pm.EndPosition;
				StayOnGround();
				return;
			}

			StepMove();
		}
		finally
		{

			// Now pull the base velocity back out.   Base velocity is set if you are on a moving object, like a conveyor (or maybe another monster?)
			pl.Velocity -= pl.BaseVelocity;
		}

		StayOnGround();

		pl.Velocity = pl.Velocity.Normal * MathF.Min( pl.Velocity.Length, (GetWishSpeed() * pl.Scale.Length) );
	}

	public virtual void StepMove()
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		MoveHelper mover = new MoveHelper( pl.Position, pl.Velocity );
		mover.Trace = mover.Trace.Size( mins, maxs );//.Ignore( pl );

		mover.TryMoveWithStep( Time.Delta, StepSize * pl.Scale.Length );

		pl.Position = mover.Position;
		pl.Velocity = mover.Velocity;
	}

	/// <summary>
	/// Add our wish direction and speed onto our velocity
	/// </summary>
	public virtual void Accelerate( Vector3 wishdir, float wishspeed, float speedLimit, float acceleration )
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		// This gets overridden because some games (CSPort) want to allow dead (observer) players
		// to be able to move around.
		// if ( !CanAccelerate() )
		//     return; 
		speedLimit *= pl.Scale.Length;
		acceleration /= pl.Scale.Length;
		if ( speedLimit > 0 && wishspeed > speedLimit )
			wishspeed = speedLimit;

		// See if we are changing direction a bit
		var currentspeed = pl.Velocity.Dot( wishdir );

		// Reduce wishspeed by the amount of veer.
		var addspeed = wishspeed - currentspeed;

		// If not going to add any speed, done.
		if ( addspeed <= 0 )
			return;

		// Determine amount of acceleration.
		var accelspeed = (acceleration * pl.Scale.Length) * Time.Delta * wishspeed * SurfaceFriction;

		// Cap at addspeed
		if ( accelspeed > addspeed )
			accelspeed = addspeed;

		pl.Velocity += wishdir * accelspeed;
	}

	/// <summary>
	/// Remove ground friction from velocity
	/// </summary>
	public virtual void ApplyFriction( float frictionAmount = 1.0f )
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		// If we are in water jump cycle, don't apply friction
		//if ( player->m_flWaterJumpTime )
		//   return; 
		// Not on ground - no friction   
		// Calculate speed 
		var speed = pl.Velocity.Length;
		if ( speed < 0.1f ) return;

		// Bleed off some speed, but if we have less than the bleed
		//  threshold, bleed the threshold amount.
		float control = (speed < StopSpeed * pl.Scale.Length) ? (StopSpeed * pl.Scale.Length) : speed;

		// Add the amount to the drop amount.
		var drop = control * Time.Delta * frictionAmount;

		// scale the velocity
		float newspeed = speed - drop;
		if ( newspeed < 0 ) newspeed = 0;

		if ( newspeed != speed )
		{
			newspeed /= speed;
			pl.Velocity *= newspeed;
		}

		// mv->m_outWishVel -= (1.f-newspeed) * mv->m_vecVelocity;
	}

	public virtual void CheckDuck()
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		bool wants = Input.Down( "Duck" );

		if ( wants != IsDucking )
		{
			if ( wants ) TryDuck();
			else TryUnDuck();
		}

		if ( IsDucking )
		{
			var delta = DuckAmount;
			DuckAmount = DuckAmount.LerpTo( (EyeHeight - DuckEyeHeight) * -1, 8 * Time.Delta );
			delta -= DuckAmount;
			SetTag( "ducked" );
			if ( pl.GroundObject is null )
			{
				pl.Position += Vector3.Up * (delta * pl.Scale);
				var pm = TraceBBox( pl.Position, pl.Position );
				if ( pm.StartedSolid )
				{
					pl.Position -= Vector3.Up * (delta * pl.Scale);
				}
			}
			FixPlayerCrouchStuck( true );
			CategorizePosition( false );
		}
		else
		{
			var delta = DuckAmount;
			DuckAmount = DuckAmount.LerpTo( 0, 8 * Time.Delta );
			delta -= DuckAmount;

			if ( pl.GroundObject is null )
			{
				pl.Position += Vector3.Up * (delta * pl.Scale);

				var pm = TraceBBox( pl.Position, pl.Position );
				if ( pm.StartedSolid )
				{
					pl.Position -= Vector3.Up * (delta * pl.Scale);
				}
			}
			CategorizePosition( false );
		}
		pl.EyeLocalPosition = pl.EyeLocalPosition.WithZ( EyeHeight + (DuckAmount) );

	}

	public virtual void TryUnDuck()
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		UpdateBBox( -1 );
		var pm = TraceBBox( pl.Position, pl.Position, DuckHeight - 4 );
		if ( pm.StartedSolid )
		{
			UpdateBBox();
			return;
		}
		IsDucking = false;
	}

	public virtual void FixPlayerCrouchStuck( bool upward )
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		int direction = upward ? 1 : 0;

		var trace = TraceBBox( pl.Position, pl.Position );
		if ( trace.GameObject == null )
			return;

		var test = pl.Position;
		for ( int i = 0; i < (DuckHeight - 4); i++ )
		{
			var org = pl.Position;
			org.z += direction;

			pl.Position = org;
			trace = TraceBBox( pl.Position, pl.Position );
			if ( trace.GameObject == null )
				return;
		}

		pl.Position = test;
	}

	public virtual void WaterSimulate()
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		if ( pl.WaterLevel > 0.4 )
		{
			CheckWaterJump();
		}

		// If we are falling again, then we must not trying to jump out of water any more.
		if ( (pl.Velocity.z < 0.0f) && IsJumpingFromWater )
		{
			WaterJumpTime = 0.0f;
		}

		// Was jump button pressed?
		if ( Input.Down( "Jump" ) )
		{
			CheckJumpButton();
		}
		SetTag( "swimming" );
	}

	public virtual float WaterJumpHeight => 8;
	protected void CheckWaterJump()
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		// Already water jumping.
		if ( IsJumpingFromWater )
			return;

		// Don't hop out if we just jumped in
		// only hop out if we are moving up
		if ( pl.Velocity.z < -180 )
			return;

		// See if we are backing up
		var flatvelocity = pl.Velocity.WithZ( 0 );

		// Must be moving
		var curspeed = flatvelocity.Length;
		flatvelocity = flatvelocity.Normal;

		// see if near an edge
		var flatforward = pl.Rotation.Forward.WithZ( 0 ).Normal;

		// Are we backing into water from steps or something?  If so, don't pop forward
		if ( curspeed != 0 && Vector3.Dot( flatvelocity, flatforward ) < 0 )
			return;

		var vecStart = pl.Position + (mins + maxs) * .5f;
		var vecEnd = vecStart + flatforward * 24;

		var tr = TraceBBox( vecStart, vecEnd );
		if ( tr.Fraction == 1 )
			return;

		vecStart.z = pl.Position.z + EyeHeight + WaterJumpHeight;
		vecEnd = vecStart + flatforward * 24;
		WaterJumpVelocity = tr.Normal * -50;

		tr = TraceBBox( vecStart, vecEnd );
		if ( tr.Fraction < 1.0 )
			return;

		// Now trace down to see if we would actually land on a standable surface.
		vecStart = vecEnd;
		vecEnd.z -= 1024;

		tr = TraceBBox( vecStart, vecEnd );
		if ( tr.Fraction < 1 && tr.Normal.z >= 0.7f )
		{
			pl.Velocity = pl.Velocity.WithZ( 256 * pl.Scale.Length );
			GameObject.Tags.Add( "waterjump" );
			WaterJumpTime = 2000;
		}
	}

	}

			LastNonZeroWishLadderVelocity = wishvel;
		}
		if ( TryLatchNextTickCounter > 0 ) wishvel = LastNonZeroWishLadderVelocity * -1;
		wishvel *= pl.ViewAngles.WithPitch( 0 ).ToRotation();
		wishvel = wishvel.Normal;


		if ( IsTouchingLadder )
		{
			if ( Input.Pressed( "Jump" ) )
			{
				var sidem = (Math.Abs( pl.ViewAngles.ToRotation().Forward.Abs().z - 1 ) * 3).Clamp( 0, 1 );
				var upm = pl.ViewAngles.ToRotation().Forward.z;

				var Eject = new Vector3();

				Eject.x = LadderNormal.x * sidem;
				Eject.y = LadderNormal.y * sidem;
				Eject.z = (3 * upm).Clamp( 0, 1 );

				pl.Velocity += (Eject * 180.0f) * pl.Scale;
				IsTouchingLadder = false;

				return;

			}
			else if ( pl.GroundObject != null && LadderNormal.Dot( wishvel ) > 0 )
			{
				IsTouchingLadder = false;

				return;
			}
		}

		const float ladderDistance = 1.0f;
		var start = pl.Position;
		Vector3 end = start + (IsTouchingLadder ? (LadderNormal * -1.0f) : wishvel) * ladderDistance;

		var pm = GameManager.ActiveScene.Trace.Ray( start, end )
					.Size( mins, maxs )
					.WithTag( "ladder" )
					//.Ignore( pl )
					.Run();

		IsTouchingLadder = false;

		if ( pm.Hit )
		{
			IsTouchingLadder = true;
			LadderNormal = pm.Normal;
		}
	}
	public virtual void LadderMove()
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		var velocity = WishVelocity;
		float normalDot = velocity.Dot( LadderNormal );

		var cross = LadderNormal * normalDot;
		pl.Velocity = (velocity - cross) + (-normalDot * LadderNormal.Cross( Vector3.Up.Cross( LadderNormal ).Normal ));

		Move();
	}

	public virtual void CategorizePosition( bool bStayOnGround )
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		SurfaceFriction = 1.0f;

		// Doing this before we move may introduce a potential latency in water detection, but
		// doing it after can get us stuck on the bottom in water if the amount we move up
		// is less than the 1 pixel 'threshold' we're about to snap to.	Also, we'll call
		// this several times per frame, so we really need to avoid sticking to the bottom of
		// water on each call, and the converse case will correct itself if called twice.
		//CheckWater();

		var point = pl.Position - Vector3.Up * (2 * pl.Scale);
		var vBumpOrigin = pl.Position;

		//
		//  Shooting up really fast.  Definitely not on ground trimed until ladder shit
		//
		bool bMovingUpRapidly = pl.Velocity.z > MaxNonJumpVelocity;
		bool bMovingUp = pl.Velocity.z > 0;

		bool bMoveToEndPos = false;

		if ( pl.GroundObject != null ) // and not underwater
		{
			bMoveToEndPos = true;
			point.z -= StepSize * pl.Scale.Length;
		}
		else if ( bStayOnGround )
		{
			bMoveToEndPos = true;
			point.z -= StepSize * pl.Scale.Length;
		}

		if ( bMovingUpRapidly || Swimming ) // or ladder and moving up
		{
			ClearGroundObject();
			return;
		}

		var pm = TraceBBox( vBumpOrigin, point, 4.0f );

		if ( pm.GameObject == null || Vector3.GetAngle( Vector3.Up, pm.Normal ) > GroundAngle )
		{
			ClearGroundObject();
			bMoveToEndPos = false;

			if ( pl.Velocity.z > 0 )
				SurfaceFriction = 0.25f;
		}
		else
		{
			UpdateGroundObject( pm );
		}

		if ( bMoveToEndPos && !pm.StartedSolid && pm.Fraction > 0.0f && pm.Fraction < 1.0f )
		{
			pl.Position = pm.EndPosition;
		}

	}

	public virtual void UpdateGroundObject( SceneTraceResult tr )
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		GroundNormal = tr.Normal;

		// VALVE HACKHACK: Scale this to fudge the relationship between vphysics friction values and player friction values.
		// A value of 0.8f feels pretty normal for vphysics, whereas 1.0f is normal for players.
		// This scaling trivially makes them equivalent.  REVISIT if this affects low friction surfaces too much.
		SurfaceFriction = tr.Surface.Friction * 1.25f;
		if ( SurfaceFriction > 1 ) SurfaceFriction = 1;

		//if ( tr.pl == GroundObject ) return;

		Vector3 oldGroundVelocity = default;
		if ( pl.GroundObject != null ) oldGroundVelocity = Vector3.Zero; //pl.GroundObject.Velocity;

		bool wasOffGround = pl.GroundObject == null;

		pl.GroundObject = tr.GameObject;

		if ( pl.GroundObject != null )
		{
			pl.BaseVelocity = Vector3.Zero; //pl.GroundObject.Velocity;
		}
	}

	/// </summary>

	}

	/// <summary>
	/// Traces the bbox and returns the trace result.
	/// LiftFeet will move the start position up by this amount, while keeping the top of the bbox at the same 
	/// position. This is good when tracing down because you won't be tracing through the ceiling above.
	/// </summary>
	public virtual SceneTraceResult TraceBBox( Vector3 start, Vector3 end, Vector3 mins, Vector3 maxs, float liftFeet = 0.0f )
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		if ( liftFeet > 0 )
		{
			liftFeet *= pl.Scale.Length;
			start += Vector3.Up * liftFeet;
			maxs = maxs.WithZ( maxs.z - liftFeet );
		}

		var tr = GameManager.ActiveScene.Trace.Ray( start + TraceOffset, end + TraceOffset )
					.Size( mins, maxs )
					.WithAnyTags( "solid", "playerclip", "passbullets", "player" )
					//.Ignore( pl )
					.Run();

		tr.EndPosition -= TraceOffset;
		return tr;
	}

	/// <summary>
	/// Try to keep a walking player on the ground when running down slopes etc
	/// </summary>
	public virtual void StayOnGround()
	{
		var pl = GameObject.Parent.Components.Get<PlayerComponent>();
		var start = pl.Position + Vector3.Up * (2 * pl.Scale);
		var end = pl.Position + Vector3.Down * (StepSize * pl.Scale);

		// See how far up we can go without getting stuck
		var trace = TraceBBox( pl.Position, start );
		start = trace.EndPosition;

		// Now trace down from a known safe position
		trace = TraceBBox( start, end );

		if ( trace.Fraction <= 0 ) return;
		if ( trace.Fraction >= 1 ) return;
		if ( trace.StartedSolid ) return;
		if ( Vector3.GetAngle( Vector3.Up, trace.Normal ) > GroundAngle ) return;

		// This is incredibly hacky. The real problem is that trace returning that strange value we can't network over.
		// float flDelta = fabs( mv->GetAbsOrigin().z - trace.m_vEndPos.z );
		// if ( flDelta > 0.5f * DIST_EPSILON )

		pl.Position = trace.EndPosition;
	}
